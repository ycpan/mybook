# 日常办公
.zshrc 配置
```bash
#开启颜色
autoload -U colors && colors

#export DJANGO_SETTINGS_MODULE=turbine.settings
GIT_PS1_SHOWUPSTREAM="auto"
GIT_PS1_SHOWCOLORHINTS="yes"
source ~/.git-prompt.sh # https://github.com/git/git/blob/master/contrib/completion/git-prompt.sh
setopt PROMPT_SUBST
#PS1='[%n@%m %c$(__git_ps1 " (%s)")]\$ '
PS1='%T%d$(__git_ps1 " (%s)")\$ '
#配置提示符模式。。 其实配置之前的PS1也是可以的 但是为了尊重说明文档。。。
PROMPT="%{$fg[red]%}%n%{$reset_color%}$(__git_ps1 " (%s)")%#>"
#在行末显示上一命令的返回状态
RPROMPT="[%{$fg_bold[yellow]%}%?%{$reset_color%}]"
export CLICOLOR=1
# ls出现的颜色
export LSCOLORS=gafacadabaegedabagacad
```
